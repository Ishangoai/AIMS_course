import os
import tempfile
import zipfile
from PIL import Image
from gradioapp.utils.image_assignment_utils import transform_image

def transform_folder(folder, grayscale_or_not, brightness, contrast_factor, degree, threshold):
    if not folder:
        return "No folder uploaded", None

    output_dir = tempfile.mkdtemp()
    processed_files = []

    for file_data in folder:
        file_path = getattr(file_data, "name", None)
        if not file_path or not os.path.exists(file_path):
            print(f"⚠️ Fichier introuvable ou non accessible : {file_data}")
            continue

        if not file_path.lower().endswith((".png", ".jpg", ".jpeg")):
            # ignore non-image
            continue

        try:
            with Image.open(file_path) as img:
                img_transformed = transform_image(img, grayscale_or_not, brightness, contrast_factor, degree, threshold)

                # Détecte le format de sortie souhaité d'après l'extension d'origine
                base_name = os.path.basename(file_path)
                name, ext = os.path.splitext(base_name)
                ext = ext.lower()

                # Si l'image a un canal alpha (RGBA/LA) et l'ext est jpeg -> convertir en RGB
                if ext in (".jpg", ".jpeg"):
                    if img_transformed.mode in ("RGBA", "LA", "P"):
                        img_to_save = img_transformed.convert("RGB")
                        output_path = os.path.join(output_dir, name + ext)
                        print(f"Conversion RGBA->RGB pour {base_name} avant sauvegarde JPEG.")
                    else:
                        img_to_save = img_transformed
                        output_path = os.path.join(output_dir, name + ext)
                else:
                    # pour PNG on peut garder la transparence si elle existe
                    img_to_save = img_transformed
                    output_path = os.path.join(output_dir, name + ext)

                # Enregistre l'image (choix de paramètres possibles selon format)
                # Pour JPEG, tu peux ajouter quality=95 si tu veux
                save_kwargs = {}
                if output_path.lower().endswith((".jpg", ".jpeg")):
                    save_kwargs["quality"] = 95

                img_to_save.save(output_path, **save_kwargs)
                processed_files.append(output_path)

        except Exception as e:
            print(f"Erreur lors du traitement de {file_path}: {e}")
            continue

    if not processed_files:
        return "Aucune image valide trouvée dans le dossier.", None

    # Zip des fichiers transformés
    zip_path = os.path.join(output_dir, "transformed_images.zip")
    with zipfile.ZipFile(zip_path, "w") as zipf:
        for f in processed_files:
            zipf.write(f, os.path.basename(f))

    return f"{len(processed_files)} images transformées.", zip_path
